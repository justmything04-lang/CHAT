# ---------- Sheet-write queue + worker (safe, batched, rate-limited) ----------
from collections import deque
import time

# Config (can be overridden in env)
SHEET_WRITE_BATCH_SIZE = int(os.getenv("SHEET_WRITE_BATCH_SIZE", "50"))   # rows per append_rows call
SHEET_WRITE_BATCH_DELAY = float(os.getenv("SHEET_WRITE_BATCH_DELAY", "1.0"))  # seconds between sheet API calls
SHEET_WORKER_SLEEP = float(os.getenv("SHEET_WORKER_SLEEP", "0.5"))  # sleep between worker loop iterations

_sheet_write_queue = deque()   # items: {"mode":"offline"|"online", "date": "YYYY-MM-DD"}
_sheet_worker_running = False
_sheet_worker_thread = None
_sheet_worker_lock = threading.Lock()

def enqueue_sheet_write(mode: str, date_str: str):
    """Add a task to write absentees/<mode>/dates/<date> -> corresponding Google Sheet."""
    _sheet_write_queue.append({"mode": mode, "date": str(date_str)})

def _process_one_sheet_task(task):
    """Read Firestore doc for task and write into the proper Google Sheet tab (batched)."""
    mode = task.get("mode")
    date_str = task.get("date")
    try:
        fs_db = firestore.client() if hasattr(firestore, "client") else None
    except Exception:
        fs_db = None

    if not fs_db:
        print("⚠️ Sheet-worker: Firestore client unavailable; skipping sheet write for", task)
        return

    # read absentee rows from Firestore
    try:
        doc_ref = fs_db.collection("absentees").document(mode).collection("dates").document(str(date_str))
        doc = doc_ref.get()
        if not doc.exists:
            print(f"⚠️ Sheet-worker: no firestore doc for absentees/{mode}/dates/{date_str}")
            rows = []
        else:
            data = doc.to_dict() or {}
            rows = data.get("rows", [])  # expecting list of {"Name","Reg ID","Date"}
    except Exception as e:
        print("⚠️ Sheet-worker: error reading firestore doc:", e)
        rows = []

    # pick target sheet id and gspread file
    sheet_id = ABSENTEE_SHEET_ID if mode == "offline" else ONLINE_ABSENTEE_SHEET_ID
    if not sheet_id:
        print(f"⚠️ Sheet-worker: no sheet id configured for mode={mode}")
        return

    # open target spreadsheet and ensure worksheet for date exists
    try:
        file = client.open_by_key(sheet_id)
        ws_title = f"{date_str}-{mode}"
        try:
            ws = file.worksheet(ws_title)
        except gspread.exceptions.WorksheetNotFound:
            # create and set header
            ws = file.add_worksheet(title=ws_title, rows="500", cols="3")
            try:
                ws.update("A1:C1", [["Name", "Reg ID", "Date"]])
            except Exception:
                pass
        # clear previous data rows (A2:C)
        try:
            ws.batch_clear(["A2:C"])
        except Exception:
            pass
    except Exception as e:
        print("⚠️ Sheet-worker: failed to open / prepare spreadsheet:", e)
        return

    # if no rows, nothing to append; finish
    if not rows:
        print(f"Sheet-worker: no rows to write for {mode}/{date_str}")
        return

    # prepare rows (list of lists)
    row_lists = []
    for r in rows:
        name = r.get("Name", "") if isinstance(r, dict) else (r[0] if len(r) > 0 else "")
        rid = r.get("Reg ID", "") if isinstance(r, dict) else (r[1] if len(r) > 1 else "")
        date_val = r.get("Date", "") if isinstance(r, dict) else (r[2] if len(r) > 2 else date_str)
        row_lists.append([name, rid, date_val])

    # Batch append with rate limiting
    try:
        idx = 0
        total = len(row_lists)
        while idx < total:

                except Exception as e2:
                    print("⚠️ Sheet-worker: append_rows retry failed:", e2)
                    break
            idx += len(batch)
            # throttle between batches
            time.sleep(SHEET_WRITE_BATCH_DELAY)
        print(f"Sheet-worker: wrote {total} rows to {ws.title}")
    except Exception as e:
        print("⚠️ Sheet-worker: unexpected error while batching:", e)

def _sheet_worker_loop():
    global _sheet_worker_running
    _sheet_worker_running = True
    try:
        while True:
            if _sheet_write_queue:
                task = _sheet_write_queue.popleft()
                _process_one_sheet_task(task)
            else:
                time.sleep(SHEET_WORKER_SLEEP)
    except Exception as e:
        print("⚠️ Sheet-worker crashed:", e)
    finally:
        _sheet_worker_running = False

def start_sheet_worker_once():
    """Start background sheet-writing worker thread if not already running."""
    global _sheet_worker_thread, _sheet_worker_running
    with _sheet_worker_lock:
        if _sheet_worker_running:
            return
        try:
            _sheet_worker_thread = threading.Thread(target=_sheet_worker_loop, daemon=True)
            _sheet_worker_thread.start()
            print("✅ Sheet-worker started (daemon thread).")
        except Exception as e:
            print("⚠️ Failed to start sheet-worker thread:", e)

def _days_since(start_date, today_date):
    return (today_date - start_date).days

def _current_biweekly_window(today_date, start_date):
    """
    Window length = 14 days. Returns (win_start, win_end, is_boundary_today).
    Cycle n: [start+14n, start+14n+13]; boundary day is win_end.
    """
    days = _days_since(start_date, today_date)
    if days < 0:
        return (today_date, today_date, False)
    idx = days // 14
    win_start = start_date + timedelta(days=14*idx)
    win_end   = win_start + timedelta(days=13)
    return (win_start, win_end, today_date == win_end)

def _collect_absences_between_with_dates(sheet_key, suffix, student_rows, win_start, win_end):
    """
    Reads absentee tabs 'YYYY-MM-DD-<suffix>' within [win_start..win_end].
    Returns: abs_map {rid: absent_days}, classes count, abs_dates {rid: [YYYY-MM-DD,...]}
    """
    try:
        f = client.open_by_key(sheet_key)
        tabs = f.worksheets()
    except Exception:
        tabs = []

    day_tabs = []
    for ws in tabs:
        t = ws.title
        if not t.endswith(suffix):
            continue
        ds = t[:-len(suffix)]
        d  = _safe_parse_date(ds)
        if d and (win_start <= d <= win_end):
            day_tabs.append((d, ds, ws))

    classes = len(day_tabs)
    ids = { str(r.get("Reg ID","")).strip() for r in student_rows }
    abs_map   = { rid: 0 for rid in ids }
    abs_dates = { rid: [] for rid in ids }

    for _, ds, ws in day_tabs:
        try:
            rows = _gs_read(lambda: ws.get_all_records())
            for a in rows:
                rid = str(a.get("Reg ID","")).strip()
                if rid in abs_map:
                    abs_map[rid] += 1
                    abs_dates[rid].append(ds)
        except Exception:
            pass

    return abs_map, classes, abs_dates

def _band_for(pct):
    if pct >= 1.0:  return "100%"
    if pct >= 0.80: return "High"
    if pct >= 0.60: return "Average"
    return "Low"

def _build_mode_summary(mode_label, student_rows, abs_map, classes):
    """
    detailed_rows = [(RegID, Name, Present, Absent, PctFloat, Band)]
    """
    detailed = []
    for r in student_rows:
        rid  = str(r.get("Reg ID","")).strip()
        name = r.get("Name","")
        absd = abs_map.get(rid, 0)
        pres = max(classes - absd, 0)
        pct  = (pres / classes) if classes else 0.0
        band = _band_for(pct)
        detailed.append((rid, name, pres, absd, pct, band))

    band_counts = {"100%":0, "High":0, "Average":0, "Low":0}
    for _, _, _, _, _, band in detailed:
        band_counts[band] += 1

    lines = []
    lines.append(f"📊 {mode_label} (last 14 days)")
    lines.append(f"Classes in window: {classes}")
    lines.append("")
    if band_counts["100%"]>0: lines.append(f"✅ 100%: {band_counts['100%']}")
    lines.append(f"🟢 High (≥80%): {band_counts['High']}")
    lines.append(f"🟡 Average (60–79%): {band_counts['Average']}")
    lines.append(f"🔴 Low (<60%): {band_counts['Low']}")
    return "\n".join(lines), detailed, band_counts

def _sorted_by_band(detailed_rows):
    # Sort: Low -> Average -> High -> 100%, then by % asc, then name
    order = {"Low":0, "Average":1, "High":2, "100%":3}
    return sorted(detailed_rows, key=lambda t: (order.get(t[5], 9), t[4], t[1].lower()))

def _days_since(start_date, today_date):
    return (today_date - start_date).days

def _current_biweekly_window(today_date, start_date):
    """
    Window length = 14 days. Returns (win_start, win_end, is_boundary_today).
    Cycle n: [start+14n, start+14n+13]; boundary day is win_end.
    """
    days = _days_since(start_date, today_date)
    if days < 0:
        return (today_date, today_date, False)
    idx = days // 14
    win_start = start_date + timedelta(days=14*idx)
    win_end   = win_start + timedelta(days=13)
    return (win_start, win_end, today_date == win_end)

def _collect_absences_between_with_dates(sheet_key, suffix, student_rows, win_start, win_end):
    """
    Reads absentee documents from Firestore absentees/<mode>/dates/<YYYY-MM-DD> within [win_start..win_end].
    Returns: abs_map {rid: absent_days}, classes count, abs_dates {rid: [YYYY-MM-DD,...]}
    """
    fs_db = firestore.client()
    # determine mode from suffix
    mode = "offline" if "offline" in suffix else "online"
    try:
        col = fs_db.collection("absentees").document(mode).collection("dates")
        docs = list(col.list_documents())
    except Exception:
        docs = []

    day_docs = []
    for d in docs:
        ds = d.id
        dt = _safe_parse_date(ds)
        if dt and (win_start <= dt <= win_end):
            day_docs.append((dt, ds, d))

    classes = len(day_docs)
    ids = { str(r.get("Reg ID","")).strip() for r in student_rows }

def _month_window_for(date_obj):
    y, m = date_obj.year, date_obj.month
    first = datetime(y, m, 1, tzinfo=ZoneInfo(TIMEZONE)).date()

def _month_window_for(date_obj):
    y, m = date_obj.year, date_obj.month
    first = datetime(y, m, 1, tzinfo=ZoneInfo(TIMEZONE)).date()
    last_day = calendar.monthrange(y, m)[1]
    last = datetime(y, m, last_day, tzinfo=ZoneInfo(TIMEZONE)).date()
    return first, last

def _fs_client():
    try:
        return firestore.client()
    except Exception:
        # fallback if a different client variable is used in your project
        try:
            return db  # sometimes `db` is the firestore client in your codebase
        except Exception:
            raise

def _fs_doc_exists(path_parts):
    """
    path_parts: list like ["absentees","offline","2025-11-01"]
    """
    fs = _fs_client()
    # join into collection/doc pattern: coll/doc/coll/doc...
    try:
        ref = fs
        # alternate traversal depending on client implementation
        if hasattr(fs, "collection"):
            # proper firestore client
            ref = fs.collection(path_parts[0])
            for i, part in enumerate(path_parts[1:]):
                if i % 1 == 0 and (i + 1) < len(path_parts[1:]) + 1:
                    # treat alternation as doc -> next collection
                    # simpler: alternate collection/doc starting from collection
                    try:
                        ref = ref.document(part)
                    except Exception:
                        ref = ref.document(part)
                    # next loop will call collection on this doc if needed
                else:
                    try:
                        ref = ref.document(part)
                    except Exception:
                        ref = ref.document(part)
        else:
            # if fs is a wrapper/dict-like
            return False
        # ultimately try get()
        if hasattr(ref, "get"):
            d = ref.get()
            return d.exists if hasattr(d, "exists") else bool(d)
        return False
    except Exception:
        return False

def _fs_get_absentee_docs(mode, win_start, win_end):
    """
    mode: "offline" or "online"
    returns list of tuples: (date_str, list_of_rows) where each row is dict with Reg ID, Name, Date
    """
    fs = _fs_client()
    out = []
    try:
        coll = fs.collection("absentees").document(mode).collection("by_date")
        # Query: date field between win_start and win_end
        start_s = str(win_start)
        end_s = str(win_end)
        q = coll.where("date_str", ">=", start_s).where("date_str", "<=", end_s)
        docs = q.stream()
        for d in docs:
            try:
                rec = d.to_dict() or {}
                date_str = rec.get("date_str") or d.id
                rows = rec.get("rows") or []
                out.append((date_str, rows))
            except Exception:
                pass
    except Exception:
        # fallback: try listing all and filter client-side
        try:
            coll = fs.collection("absentees").document(mode).collection("by_date")
            docs = coll.stream()
            for d in docs:
                rec = d.to_dict() or {}
                date_str = rec.get("date_str") or d.id
                try:
                    dt = _safe_parse_date(date_str)
                except Exception:
                    dt = None
                if dt and (win_start <= dt <= win_end):
                    out.append((date_str, rec.get("rows") or []))
        except Exception:
            pass
    # sort by date
    try:
        out.sort(key=lambda x: x[0])
    except Exception:
        pass
    return out

def _fs_write_report(mode, collection_name, doc_id, payload):
    """
    Write computed report to `reports/{collection_name}/{doc_id}` as a document.
    payload is a dict; this writes timestamp 'created_at' and 'rows' if present.
    """
    fs = _fs_client()
    try:
        docref = fs.collection("reports").document(collection_name).collection("items").document(str(doc_id))
        payload = dict(payload)
        payload.setdefault("created_at", datetime.now(ZoneInfo(TIMEZONE)).isoformat())
        docref.set(payload)
        return True
    except Exception:
        try:
            # fallback to top-level path
            docref = fs.collection("reports").document(str(doc_id))
            docref.set(payload)
            return True
        except Exception:
            return False

def _make_unique_report_title(base_title, report_type):
    """
    Ensure uniqueness by checking `reports/{report_type}/items` for existing doc IDs.
    Returns a safe title (string) to use as doc id.
    """
    fs = _fs_client()
    try:
        coll = fs.collection("reports").document(report_type).collection("items")
        # direct check
        existing = [d.id for d in coll.stream()]
    except Exception:
        existing = []
    if base_title not in existing:
        return base_title
    suffix = datetime.now(ZoneInfo(TIMEZONE)).strftime("%Y%m%d%H%M%S")
    return f"{base_title} {suffix}"

def _collect_absences_between(sheet_key, suffix, student_rows, win_start, win_end):
    """
    Now reads absentees from Firestore. sheet_key/suffix are ignored; student_rows used for id set.
    Returns: abs_map {rid: absent_days}, classes count, abs_dates {rid: [YYYY-MM-DD,...]}
    """
    # identify mode from suffix
    mode = "offline" if suffix and "offline" in suffix else "online"
    day_docs = _fs_get_absentee_docs(mode, win_start, win_end)  # list of (date_str, rows)
    classes = len(day_docs)
    ids = { str(r.get("Reg ID","")).strip() for r in student_rows }
    abs_map = { rid: 0 for rid in ids }
    abs_dates = { rid: [] for rid in ids }

    for ds, rows in day_docs:
        try:
            for a in rows:
                rid = str(a.get("Reg ID","")).strip() if isinstance(a, dict) else str(a[1]).strip() if isinstance(a, (list,tuple)) and len(a) > 1 else ""
                if rid in abs_map:
                    abs_map[rid] += 1
                    abs_dates[rid].append(ds)
        except Exception:
            pass
    return abs_map, classes, abs_dates

def _risk_band_for(pct, risk_threshold):
    return "Risk" if pct <= risk_threshold else None

def _write_monthly_sheet(win_start, win_end, off_rows, on_rows):
    """
    Writes monthly report into Firestore instead of a sheet tab.
    Returns (doc_id, title) as replacement for gid/title.
    """
    base_title = f"MON {win_start.strftime('%Y-%m')}"
    title = _make_unique_report_title(base_title, "monthly")
    rows = []
    for r in _sorted_by_band(off_rows):
        rid, name, pres, absd, pct, band = r
        rows.append({
            "mode": "Offline",
            "regid": rid,
            "name": name,
            "present": pres,
            "absent": absd,
            "percent": round(pct*100,1),
            "band": band,
            "window_start": str(win_start),
            "window_end": str(win_end),
            "created_at": now_ts()
        })
    for r in _sorted_by_band(on_rows):
        rid, name, pres, absd, pct, band = r
        rows.append({
            "mode": "Online",
            "regid": rid,
            "name": name,
            "present": pres,
            "absent": absd,
            "percent": round(pct*100,1),
            "band": band,
            "window_start": str(win_start),
            "window_end": str(win_end),
            "created_at": now_ts()
        })
    payload = {
        "title": title,
        "window_start": str(win_start),
        "window_end": str(win_end),
        "rows": rows
    }
    ok = _fs_write_report("monthly", "monthly", title, payload)
    if ok:
        return title, title
    return None, None

def _post_public(msg_text):
    group_id = os.getenv("REPORT_GROUP_ID","").strip()
    topic_id = os.getenv("REPORT_TOPIC_ID","").strip()
    if not group_id or not topic_id:
        print("⚠️ REPORT_GROUP_ID/REPORT_TOPIC_ID not set; skipping public topic post.")
        return
    try:
        bot.send_message(chat_id=int(group_id), text=_truncate_text(msg_text), message_thread_id=int(topic_id))
    except Exception as e:
        print("⚠️ Public post failed:", e)

def send_monthly_report():
    """
    Auto/manual monthly report. Uses Firestore for reports & absentee storage.
    """
    now_local = datetime.now(ZoneInfo(TIMEZONE)).date()
    m_first, m_last = _month_window_for(now_local)
    win_end = m_last if now_local == m_last else now_local
    win_start = m_first
    risk_threshold = float(os.getenv("RISK_THRESHOLD","0.50"))

    off_students = get_cached_master_list()
    on_students  = get_cached_online_master_list()
    off_absmap, off_classes, off_absdates = _collect_absences_between(None, "-offline", off_students, win_start, win_end)
    on_absmap,  on_classes,  on_absdates  = _collect_absences_between(None, "-online",  on_students,  win_start, win_end)

    off_block, off_detailed, _ = _build_mode_summary("Offline", off_students, off_absmap, off_classes)
    on_block,  on_detailed,  _ = _build_mode_summary("Online",  on_students,  on_absmap,  on_classes)

    def _riskify(rows):
        out = []
        for rid, name, pres, absd, pct, band in rows:
            if pct <= risk_threshold:
                out.append((rid, name, pres, absd, pct, band))
        return out
    risk_off = _riskify(off_detailed)
    risk_on  = _riskify(on_detailed)
    risk_total = len(risk_off) + len(risk_on)

    # write report to Firestore
    tab_id, tab_title = _write_monthly_sheet(win_start, win_end, off_detailed, on_detailed)
    link = f"firestore://reports/monthly/{tab_id}" if tab_id else "reports:monthly (check Firestore)"

    title = f"📅 Monthly Summary: {win_start} – {win_end}"
    public_msg = f"{title}\n\n{off_block}\n\n{on_block}\n\n🚩 Risk Watch (≤{int(risk_threshold*100)}%): {risk_total}"

    _send_parent_risk_alerts(risk_on,  on_absdates,  window_label)

    for student in get_cached_master_list():
        msg_s = APP_STUDENT_HIGH.format(
            student_name=student["Name"],
            period_type="Monthly",
            range_or_course=f"{win_start}–{win_end}"
        )
        msg_p = APP_PARENT_HIGH.format(
            student_name=student["Name"],
            period_type="Monthly",
            range_or_course=f"{win_start}–{win_end}"
        )
        safe_send_chat(student["Reg ID"], msg_s)
        if student.get("ParentChatId"):
            safe_send_chat(student["ParentChatId"], msg_p)
        try:
            db.update_streak(student["Reg ID"], boost=True)
        except Exception:
            pass

        publish_top3_to_teacher_and_topic()
    except Exception as e:
